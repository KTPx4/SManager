Create table [Admin]
(
	[ID] int IDENTITY(1,1) Primary key 
	[User] varchar(250),
	[Password] varchar(250),
	[Name] varchar(250),
	LinkAVT Text default('user.png'),					  /* Links folder store avatar */						
)

go

Create table Manager
(
	[ID] int IDENTITY(1,1) Primary key ,
	[Name] varchar(250),
	[User] varchar(250) unique,
	[Pass] varchar(250) ,	
	birthDay date,
	Phone varchar(10),
	[Status] int default(0),	/* 0 active, 1 block, -1 deleted */	
	isOnline bit default(0),
	LinkAVT Text default('user.png')	 			  /* Link folder store avatar */
		
)

go

Create table Employee
(
	[ID] int IDENTITY(1,1) Primary key,
	[Name] varchar(250),
	[User] varchar(250) unique,
	[Pass] varchar(250) ,
	birthDay date,
	Phone varchar(10),
	[Status] int default(0),	/* 0 active, 1 block, -1 deleted */
	isOnline bit default(0),
	LinkAVT Text default('user.png')	 			  /* Link folder store avatar */
				
)


go

Create table Student(	 

	[ID] varchar(250) Primary key,
	[Name] varchar(250),
	Class varchar(250),
	birthDay date,
	[Address] varchar(250),
	Phone varchar(10),
	isDelete bit,
	
)

go

Create table Certification
(		
	IDCertificate varchar(250) Primary key,
	[Name] varchar(250),
	[SID] varchar(250),
	IssueDate date,
	ExpiryDate date,
	Grade varchar(50),  /* Demonstrate for score or level of certificate */
	
		
	CONSTRAINT FK_StudentCertificate FOREIGN KEY ([SID])
    REFERENCES Student([ID])

)
go

Create table HistoryManager(
	[IDLogin] int IDENTITY(1,1) Primary key ,
	[ID] int,
	TimeLogin datetime	,
	 
	CONSTRAINT FK_HisLoginM FOREIGN KEY ([ID])
	REFERENCES Manager([ID])
)

go

Create table HistoryEmployee(
	[IDLogin] int IDENTITY(1,1) Primary key,
	[ID] int,
	TimeLogin datetime	,
	 
	CONSTRAINT FK_HisLoginE FOREIGN KEY ([ID])
	REFERENCES Employee([ID])
)
